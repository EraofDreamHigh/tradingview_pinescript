study(title="RSI_BUY", shorttitle="RSI_BUY_Huobi30m", overlay=false)
src = close, 
len = input(362, minval=1, title="RSI Length")
len2 = input(10, minval=1, title="LowerLine")
up = rma(max(change(src), 0), len)
down = rma(-min(change(src), 0), len)
rsi = down == 0 ? 100 : up == 0 ? 0 : 100 - (100 / (1 + up / down))
emaRSI = ema(rsi,len2)

plot(rsi, title="RSI", style=line, linewidth=3, color=silver)
